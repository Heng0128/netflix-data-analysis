数据清洗

去除重复记录、格式化日期、统一格式
# 去除重复记录
df.drop_duplicates(inplace=True)
print(f"去重后数据形状: {df.shape}")

# 格式化日期
df['date_added'] = pd.to_datetime(df['date_added'], format='mixed', errors='coerce')

# 提取年份
df['year_added'] = df['date_added'].dt.year

# 统一类型标签
df['type'] = df['type'].str.strip()

# 处理时长字段
df['duration_num'] = df['duration'].str.extract(r'(\d+)').astype(float)

# 删除时长缺失的记录
df.dropna(subset=['duration_num'], inplace=True)
print(f"处理后数据形状: {df.shape}")

缺失值处理

填充或删除缺失值
# 查看缺失值
print("缺失值统计:")
print(df.isnull().sum())

# 填充分类字段
df['director'].fillna('Unknown', inplace=True)
df['cast'].fillna('Unknown', inplace=True)
df['country'].fillna('Unknown', inplace=True)

# 删除日期缺失的记录
df.dropna(subset=['date_added', 'rating'], inplace=True)

# 确认处理结果
print(f"\n处理后缺失值:")
print(df.isnull().sum())

异常值检测

IQR 和 Z-score 双方法检测
# IQR 法检测异常值
Q1 = df['duration_num'].quantile(0.25)
Q3 = df['duration_num'].quantile(0.75)
IQR = Q3 - Q1
lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR
df['is_outlier_duration'] = (
    (df['duration_num'] < lower) | (df['duration_num'] > upper)
).astype(int)
print(f"IQR异常值数量: {df['is_outlier_duration'].sum()}")

# Z-score 法
from scipy import stats
df['duration_zscore'] = stats.zscore(df['duration_num'])
df['is_outlier_zscore'] = (abs(df['duration_zscore']) > 3).astype(int)
print(f"Z-score异常值数量: {df['is_outlier_zscore'].sum()}")

特征工程

提取和生成新的衍生特征
# 拆分列表字段
df['genres_list'] = df['listed_in'].str.split(', ')
df['countries_list'] = df['country'].str.split(', ')
df['cast_list'] = df['cast'].str.split(', ')

# 统计数量
df['genre_count'] = df['genres_list'].apply(len)
df['country_count'] = df['countries_list'].apply(lambda x: len(x) if isinstance(x, list) else 0)
df['cast_count'] = df['cast_list'].apply(lambda x: len(x) if isinstance(x, list) else 0)

# 提取主特征
df['primary_genre'] = df['genres_list'].apply(lambda x: x[0] if isinstance(x, list) and len(x) > 0 else 'Unknown')
df['primary_country'] = df['countries_list'].apply(lambda x: x[0] if isinstance(x, list) and len(x) > 0 else 'Unknown')
df['has_director'] = (df['director'] != 'Unknown').astype(int)

# 保存清洗后数据
df.to_csv('netflix_titles_cleaned.csv', index=False)
print(f"处理后数据形状: {df.shape}")
print(f"新增 11 个衍生字段")
